108void ResponseBuilder::handleCgiOutput(HttpResponse& response, const std::string& output)
109{
110 std::size_t sep = output.find("\r\n\r\n");
111 if (sep == std::string::npos)
112 {
113 Logger::instance().log(ERROR, "ResponseBuilder: invalid CGI output (no header separator)");
114 response.setStatusCode(ResponseStatus::BadGateway);
115 return ;
116 }
117
118 std::string headersPart = output.substr(0, sep);
119 std::string bodyPart = output.substr(sep + 4);
120
121 std::istringstream headerStream(headersPart);
122 std::string line;
123
124 while (std::getline(headerStream, line))
125 {
126 if (!line.empty() && line[line.size() - 1] == '\r')
127 line.erase(line.size() - 1);
128
129 std::size_t colon = line.find(':');
130 if (colon == std::string::npos)
131 continue ;
132
133 std::string key = trim(line.substr(0, colon));
134 std::string value = trim(line.substr(colon + 1));
135
136 response.addHeader(key, value);
137
138 if (toLower(key) == "status")
139 {
140 std::istringstream iss(value);
141 int status;
142 if (iss >> status)
143 response.setStatusCode(static_cast<ResponseStatus::code>(status));
144 }
145 }
146
147 response.appendBody(bodyPart);
148 response.addHeader("Content-Length", toString(bodyPart.size()));
149}

158void ResponseBuilder::build(ClientConnection& client, HttpRequest& req, HttpResponse& res)
159{
160 Logger::instance().log(DEBUG, "[Started] ResponseBuilder::build");
161 Logger::instance().log(DEBUG,
162 "ResponseBuilder: StatusCode -> " + toString(res.getStatusCode()));
163
164 setMinimumHeaders(res);
165 res.setReasonPhrase(res.getStatusCode());
166 res.setVersion("1.1");
167 res.addHeader("Connection", "keep-alive");
168
169 // Handle explicit connection close
170 if (req.getMeta().shouldClose())
171 {
172 res.addHeader("Connection", "close");
173 req.getMeta().setConnectionClose(true);
174 }
175
176 // Generate error page if response >= 400
177 if (res.getStatusCode() >= 400)
178 {
179 if (shouldCloseConnection(res.getStatusCode()))
180 {
181 res.addHeader("Connection", "close");
182 req.getMeta().setConnectionClose(true);
183 }
184
185 if (!errorPageConfig(client.getServerConfig().getRoot(), res, client.getServerConfig()))
186 {
187 std::string content = errorPageGenerator(res.getStatusCode());
188 handleStaticPageOutput(res, content, "text/html");
189 }
190 }
191
192 Logger::instance().log(DEBUG, "[Finished] ResponseBuilder::build");
193}

200bool ResponseBuilder::errorPageConfig(const std::string& root, HttpResponse& res, const ServerConfig& config)
201{
202 int statusCode = static_cast<int>(res.getStatusCode());
203 const std::map<int, std::string>& errorsPages = config.getErrorPage();
204
205 std::map<int, std::string>::const_iterator it = errorsPages.find(statusCode);
206 if (it == errorsPages.end())

210 std::ifstream file(path.c_str(), std::ios::binary);
211
212 if (!file)

215 res.setStatusCode(ResponseStatus::InternalServerError);
216 return (false);
217 }
218
219 std::ostringstream buffer;
220 buffer << file.rdbuf();
221 file.close();
222
223 handleStaticPageOutput(res, buffer.str(), "text/html");
224
225 Logger::instance().log(DEBUG, "ResponseBuilder: served custom error page -> " + path);
226 return (true);
227}
