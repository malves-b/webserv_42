128void WebServer::receiveRequest(size_t i)
129{
130 std::map<int, ClientConnection>::iterator it = _clients.find(_pollFDs[i].fd);
131
132 if (it == _clients.end())
133 {
134 Logger::instance().log(ERROR, "WebServer::receiveRequest: unknown client fd=" + toString(_pollFDs[i].fd));
135 return ;
136 }
137
138 ClientConnection& client = it->second;
139
140 try
141 {
142 ssize_t bytesRecv = client.recvData();
143 Logger::instance().log(DEBUG, "WebServer::receiveRequest bytesRecv=" + toString(bytesRecv));
144
145 if (bytesRecv > 0 && client.completedRequest())
146 {
147 Logger::instance().log(DEBUG, "WebServer::receiveRequest: full request received");
148 Dispatcher::dispatch(client);
149
150 if (!client.hasCgi())
151 {
152 _pollFDs[i].events = POLLOUT;
153 client.setSentBytes(0);
154 }
155 else
156 {
157 _cgiFdToClientFd[client.getCgiFd()] = client.getFD();
158 addCgiPollFd(client.getCgiFd());
159 _pollFDs[i].events = POLLIN;
160 _pollFDs[i].revents = 0;
161 }
162 }
163 else if (client.getRequest().getMeta().getExpectContinue())
164 {
165 client.setResponseBuffer("HTTP/1.1 100 Continue\r\n\r\n");
166 _pollFDs[i].events = POLLOUT;
167 client.setSentBytes(0);
168 client.getRequest().getMeta().setExpectContinue(false);
169 }
170 else if (bytesRecv == 0)
171 {
172 Logger::instance().log(INFO, "WebServer::receiveRequest: client disconnected");
173 removeClientConnection(client.getFD(), i);
174 }
175 }
176 catch (const std::exception& e)
177 {
178 Logger::instance().log(ERROR, std::string("WebServer::receiveRequest exception -> ") + e.what());
179 removeClientConnection(client.getFD(), i);
180 }

187void WebServer::sendResponse(size_t i)
188{
189 Logger::instance().log(DEBUG, "[Started] WebServer::sendResponse");
190
191 std::map<int, ClientConnection>::iterator it = _clients.find(_pollFDs[i].fd);
192
193 if (it == _clients.end())
194 {
195 Logger::instance().log(WARNING, "WebServer::sendResponse: unknown FD " + toString(_pollFDs[i].fd));
196 return ;
197 }
198
199 ClientConnection& client = it->second;
200
201 try
202 {
203 size_t totalLen = client.getResponseBuffer().length();
204 size_t sent = client.getSentBytes();
205 size_t toSend = (totalLen > sent) ? (totalLen - sent) : 0;
206
207 if (!toSend)
208 {
209 _pollFDs[i].events = POLLIN;
210 _pollFDs[i].revents = 0;
211 return ;
212 }
213
214 ssize_t bytesSent = client.sendData(client, sent, toSend);
215 if (bytesSent > 0)
216 {
217 client.setSentBytes(sent + static_cast<size_t>(bytesSent));
218
219 if (client.getSentBytes() == totalLen)
220 {
221 client.clearBuffer();
222 client.setSentBytes(0);
223 _pollFDs[i].events = POLLIN;
224
225 if (!client.getKeepAlive())
226 {
227 Logger::instance().log(INFO, "WebServer::sendResponse: closing connection (no keep-alive)");
228 removeClientConnection(it->second.getFD(), i);
229 }
230 }
231 }
232 }
233 catch (const std::exception& e)
234 {
235 Logger::instance().log(ERROR, std::string("WebServer::sendResponse exception -> ") + e.what());

331void WebServer::runServer(void)
332{
333 Logger::instance().log(INFO, "[Started] WebServer::runServer");
334
335 while (!Signals::shouldStop())
336 {
337 if (_pollFDs.empty())
338 {
339 usleep(100 * 1000); // 100ms
340 sweepCgiTimeouts();
341 continue;
342 }
343
344 int timeout = getPollTimeout();
345 int ready = ::poll(&_pollFDs[0], _pollFDs.size(), timeout);
346 sweepCgiTimeouts();
347
348 if (Signals::shouldStop())
349 break;
350
351 if (ready == -1)
352 {
353 if (errno == EINTR)
354 continue;
355 Logger::instance().log(ERROR, "poll() failed, continuing main loop");
356 continue;
357 }
358
359 for (ssize_t i = _pollFDs.size() - 1; i >= 0; --i)
360 {
361 const int fd = _pollFDs[i].fd;
362 const short re = _pollFDs[i].revents;
363 if (!re)
364 continue;
365
366 // --- CGI PIPE HANDLING ---
367 if (_cgiFdToClientFd.count(fd))
368 {
369 if (re & (POLLIN | POLLHUP | POLLRDHUP))
370 handleCgiReadable(i);
371 else if (re & (POLLERR | POLLNVAL))
372 {
373 removeCgiPollFd(fd);
374 int clientFd = _cgiFdToClientFd[fd];
375 std::map<int, ClientConnection>::iterator itc = _clients.find(clientFd);
376 if (itc != _clients.end())
377 {
378 ClientConnection& c = itc->second;
379 c.getResponse().setStatusCode(ResponseStatus::BadGateway);
380 ResponseBuilder::build(c, c.getRequest(), c.getResponse());
381 c.setResponseBuffer(ResponseBuilder::responseWriter(c.getResponse()));
382 for (size_t k = 0; k < _pollFDs.size(); ++k)
383 {
384 if (_pollFDs[k].fd == clientFd)
385 {
386 _pollFDs[k].events = POLLOUT;
387 break;
388 }
389 }
390 c.clearCgi();
391 }
392 }
393 continue;
394 }
395
396 // --- LISTEN SOCKET ---
397 if (re & POLLIN)
398 {
399 std::map<int, size_t>::iterator itSrv = _socketToServerIndex.find(fd);
400 if (itSrv != _socketToServerIndex.end()) {
401 queueClientConnections(*_serverSocket[itSrv->second]);
402 continue;
403 }
404 }
405
406 // --- CLIENT SOCKET ---
407 if (re & POLLIN)
408 {
409 std::map<int, ClientConnection>::iterator cit = _clients.find(fd);
410 if (cit != _clients.end())
411 receiveRequest(i);
412 continue;
413 }
414
415 if (re & (POLLERR | POLLHUP | POLLRDHUP | POLLNVAL))
416 {
417 std::map<int, ClientConnection>::iterator itc = _clients.find(fd);
418 if (itc != _clients.end())
419 removeClientConnection(itc->second.getFD(), i);
420 continue;
421 }
422
423 if (re & POLLOUT)
424 sendResponse(i);
425 }
426 }
427
428 gracefulShutdown();
429 Logger::instance().log(INFO, "[Finished] WebServer::runServer");
430}

472void WebServer::handleCgiReadable(int pollIndex)
473{
474 int cgiFd = _pollFDs[pollIndex].fd;
475 std::map<int,int>::iterator mapIt = _cgiFdToClientFd.find(cgiFd);
476 if (mapIt == _cgiFdToClientFd.end())
477 {
478 removeCgiPollFd(cgiFd);
479 return;
480 }
481 int clientFd = mapIt->second;
482
483 std::map<int, ClientConnection>::iterator it = _clients.find(clientFd);
484 if (it == _clients.end())
485 {
486 removeCgiPollFd(cgiFd);
487 return;
488 }
489 ClientConnection& client = it->second;
490
491 char buf[4096];
492 for (;;)
493 {
494 ssize_t n = ::read(cgiFd, buf, sizeof(buf));
495 if (n > 0)
496 {
497 client.cgiBuffer().append(buf, n);
498 continue;
499 }
500 if (n == 0)
501 {
502 int st = 0;
503 pid_t res = waitpid(client.getCgiPid(), &st, WNOHANG);
504 if (res == 0)
505 break;
506 removeCgiPollFd(cgiFd);
507 Signals::unregisterCgiProcess(client.getCgiPid());
508
509 ResponseBuilder::handleCgiOutput(client.getResponse(), client.cgiBuffer());
510 ResponseBuilder::build(client, client.getRequest(), client.getResponse());
511 client.setResponseBuffer(ResponseBuilder::responseWriter(client.getResponse()));
512
513 for (size_t i = 0; i < _pollFDs.size(); ++i)
514 {
515 if (_pollFDs[i].fd == clientFd)
516 {
517 _pollFDs[i].events = POLLOUT;
518 _pollFDs[i].revents = 0;
519 break;
520 }
521 }
522
523 client.clearCgi();
524 return;
525 }
526 if (n <= 0)
527 break;
528 }

534void WebServer::sweepCgiTimeouts()
535{
536 std::time_t now = std::time(NULL);
537 std::vector<int> toKill;
538
539 for (std::map<int,int>::iterator it = _cgiFdToClientFd.begin();
540 it != _cgiFdToClientFd.end(); ++it)
541 {
542 int cgiFd = it->first;
543 int clientFd = it->second;
544
545 std::map<int, ClientConnection>::iterator cit = _clients.find(clientFd);
546 if (cit == _clients.end())
547 {
548 toKill.push_back(cgiFd);
549 continue;
550 }
551 ClientConnection& c = cit->second;
552 double elapsed = difftime(now, c.getCgiStart());
553 if (elapsed >= Signals::CGI_TIMEOUT_SEC)
554 {
555 Logger::instance().log(WARNING, "CGI timeout, killing pid=" + toString(c.getCgiPid()));
556 kill(c.getCgiPid(), SIGKILL);
557 int st;
558 waitpid(c.getCgiPid(), &st, 0);
559 Signals::unregisterCgiProcess(c.getCgiPid());
560
561 c.getResponse().setStatusCode(ResponseStatus::GatewayTimeout);
562 ResponseBuilder::build(c, c.getRequest(), c.getResponse());
563 c.setResponseBuffer(ResponseBuilder::responseWriter(c.getResponse()));
564
565 for (size_t i = 0; i < _pollFDs.size(); ++i)
566 {
567 if (_pollFDs[i].fd == clientFd)
568 {
569 _pollFDs[i].events = POLLOUT;
570 _pollFDs[i].revents = 0;
571 break;
572 }
573 }
574
575 ::close(cgiFd);
576 toKill.push_back(cgiFd);
577 c.clearCgi();
578 }
579 }
580
581 for (size_t i = 0; i < toKill.size(); ++i)
582 removeCgiPollFd(toKill[i]);
