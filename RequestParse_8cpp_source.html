19void RequestParse::handleRawRequest(const std::string& chunk, HttpRequest& req, const ServerConfig& config)
20{
21 req.appendRaw(chunk);
22 std::string& rawRequest = req.getRaw();
23 std::string& buffer = req.getBuffer();
24 std::size_t i = 0;
25
26 if (req.getState() == RequestState::Complete)
27 return ;
28
29 while (i < rawRequest.size() && req.getState() != RequestState::Complete)
30 {
31 // Handle request line and headers
32 if (req.getState() < RequestState::Body)
33 {
34 char ch = rawRequest[i];
35
36 if (ch == '\r')
37 {
38 if (i + 1 >= rawRequest.size())
39 break ;
40
41 if (rawRequest[i + 1] != '\n')
42 {
43 req.setParseError(ResponseStatus::BadRequest);
44 return ;
45 }
46
47 if (req.getState() == RequestState::RequestLine)
48 {
49 if (!buffer.empty())
50 requestLine(buffer, req, config);
51
52 if (req.getParseError() != ResponseStatus::OK)
53 {
54 req.setRequestState(RequestState::Complete);
55 return ;
56 }
57 req.setRequestState(RequestState::Headers);
58 }
59 else
60 {
61 if (buffer.empty())
62 {
63 // End of headers: determine next state
64 if (req.getMeta().getContentLength() == 0 && !req.getMeta().isChunked())
65 req.setRequestState(RequestState::Complete);
66 else
67 req.setRequestState(RequestState::Body);
68
69 req.clearBuffer();
70 i += 2;
71 continue ;
72 }
73 else
74 {
75 headers(buffer, req, config.getClientMaxBodySize());
76 }
77 }
78 req.clearBuffer();
79 i += 2;
80 continue ;
81 }
82 buffer.push_back(ch);
83 ++i;
84 continue ;
85 }
86
87 // Handle body (Content-Length or chunked)
88 if (req.getState() == RequestState::Body)
89 {
90 body(rawRequest[i], req, config.getClientMaxBodySize());
91 ++i;
92 }
93 }
94
95 // Remove processed data from buffer
96 if (i > 0)
97 req.getRaw().erase(0, i);
98
99 // Normalize chunked requests (for CGI or POST processing)
100 if (req.getMeta().isChunked() && req.getState() == RequestState::Complete)
101 {
102 Logger::instance().log(DEBUG, "RequestParse: finalizing chunked body for CGI");
103 req.getMeta().setChunked(false);
104 req.getMeta().setContentLength(req.getBody().size());
105 req.addHeader("content-length", toString(req.getBody().size()));
106
107 if (req.hasHeader("transfer-encoding"))
108 req.removeHeader("transfer-encoding");
109 }
110
111 Logger::instance().log(DEBUG,
112 "RequestParse::handleRawRequest consumed=" + toString(i) + " remaining=" + toString(rawRequest.size()));
113}

118void RequestParse::requestLine(const std::string& buffer, HttpRequest& req, const ServerConfig& config)
119{
120 Logger::instance().log(DEBUG, "[Started] RequestParse::reqLine");
121
122 const std::vector<std::string> tokens = split(buffer, " ");
123
124 if (tokens.size() != 3)
125 {
126 req.setParseError(ResponseStatus::BadRequest);
127 return ;
128 }
129
130 uri(tokens[1], req);
131 method(tokens[0], req, config);
132
133 const std::string version = tokens[2];
134
135 if (version.size() < 8 || version.substr(0, 5) != "HTTP/" || version.find('.') == std::string::npos)
136 {
137 req.setParseError(ResponseStatus::BadRequest);
138 return ;
139 }
140
141 std::vector<std::string> parts = split(version.substr(5), ".");
142 if (parts.size() != 2)
143 {
144 req.setParseError(ResponseStatus::BadRequest);
145 return ;
146 }
147
148 if (parts[0] != "1" || parts[1] != "1")
149 {
150 req.setParseError(ResponseStatus::HttpVersionNotSupported);
151 return ;
152 }
153
154 req.setMajor(std::atoi(parts[0].c_str()));
155 req.setMinor(std::atoi(parts[1].c_str()));

157 Logger::instance().log(DEBUG, "[Finished] RequestParse::reqLine");

208void RequestParse::headers(const std::string& buffer, HttpRequest& req, std::size_t maxBodySize)
209{
210 Logger::instance().log(DEBUG, "[Started] RequestParse::headers");
211
212 std::string::size_type pos = buffer.find(":");
213 if (pos == std::string::npos)
214 {
215 req.setParseError(ResponseStatus::BadRequest);
216 req.setRequestState(RequestState::Complete);
217 Logger::instance().log(ERROR, "RequestParse::headers BadRequest (missing colon)");
218 return ;
219 }
220
221 std::string key = toLower(trim(buffer.substr(0, pos)));
222 std::string value = trim((pos + 1 < buffer.size()) ? buffer.substr(pos + 1) : std::string());
223
224 if (buffer.length() > MAX_HEADER_LINE)
225 {
226 req.setParseError(ResponseStatus::PayloadTooLarge);
227 req.setRequestState(RequestState::Complete);
228 Logger::instance().log(ERROR, "RequestParse::headers PayloadTooLarge");
229 return ;
230 }
231
232 if (key == "host")
233 req.getMeta().setHost(value);
234 else if (key == "content-length")
235 {
236 long size = std::atol(value.c_str());
237
238 if (isGreaterThanMaxBodySize(size, maxBodySize))
239 {
240 req.setParseError(ResponseStatus::PayloadTooLarge);
241 req.setRequestState(RequestState::Complete);
242 Logger::instance().log(ERROR, "RequestParse::headers Content-Length exceeds limit");
243 return ;
244 }
245 req.getMeta().setContentLength(size);
246 }
247 else if (key == "transfer-encoding")
248 {
249 std::string v = toLower(value);
250 if (v.find("chunked") != std::string::npos)
251 req.getMeta().setChunked(true);
252 else if (v != "identity")
253 {
254 req.setParseError(ResponseStatus::BadRequest);
255 req.setRequestState(RequestState::Complete);
256 Logger::instance().log(ERROR, "RequestParse::headers Unsupported transfer-encoding: " + value);
257 return ;
258 }
259 }
260 else if (key == "connection")
261 req.getMeta().setConnectionClose(toLower(value) == "close");
262 else if (key == "expect")
263 {
264 if (toLower(value) == "100-continue")
265 {
266 req.getMeta().setExpectContinue(true);
267 Logger::instance().log(DEBUG, "RequestParse::headers Expect: 100-continue");
268 }
269 else
270 {
271 req.setParseError(ResponseStatus::BadRequest);
272 req.setRequestState(RequestState::Complete);
273 Logger::instance().log(ERROR, "RequestParse::headers BadRequest on Expect header");
274 return ;
275 }
276 }
277
278 req.addHeader(key, value);
279 Logger::instance().log(DEBUG, "[Finished] RequestParse::headers");
280}

310void RequestParse::bodyChunked(char c, HttpRequest& req, std::size_t maxBodySize)
311{
312 std::string& buffer = req.getBuffer();
313 std::string& chunkBuffer = req.getChunkBuffer();
314
315 if (isGreaterThanMaxBodySize(req.getCurrentChunkSize(), maxBodySize))
316 {
317 req.setParseError(ResponseStatus::PayloadTooLarge);
318 req.setRequestState(RequestState::Complete);
319 return ;
320 }
321
322 // Waiting for CRLF separator after a chunk
323 if (req.isExpectingChunkSeparator())
324 {
325 buffer.push_back(c);
326
327 if (buffer.size() >= 2 &&
328 buffer[buffer.size() - 2] == '\r' &&
329 buffer[buffer.size() - 1] == '\n')
330 {
331 req.clearBuffer();
332 req.setExpectingChunkSeparator(false);
333 req.setParsingChunkSize(true);
334 }
335 else if (buffer.size() > 2)
336 {
337 req.setParseError(ResponseStatus::BadRequest);
338 req.setRequestState(RequestState::Complete);
339 }
340 return ;
341 }
342
343 // Parsing current chunk size
344 if (req.isParsingChunkSize())
345 {
346 buffer.push_back(c);
347
348 if (buffer.size() >= 2 &&
349 buffer[buffer.size() - 2] == '\r' &&
350 buffer[buffer.size() - 1] == '\n')
351 {
352 int size = stringToHex(buffer.substr(0, buffer.size() - 2));
353 req.clearBuffer();
354
355 if (size < 0)
356 {
357 req.setParseError(ResponseStatus::BadRequest);
358 req.setRequestState(RequestState::Complete);
359 return ;
360 }
361 if (size == 0)
362 {
363 req.setRequestState(RequestState::Complete);
364 return ;
365 }
366 req.setCurrentChunkSize(size);
367 req.setParsingChunkSize(false);
368 }
369 return ;
370 }
371
372 // Reading current chunk body
373 chunkBuffer.push_back(c);
374
375 if ((int)chunkBuffer.size() == req.getCurrentChunkSize())
376 {
377 req.appendBody(chunkBuffer);
378 req.clearChunkBuffer();
379 req.setExpectingChunkSeparator(true);
380 }
