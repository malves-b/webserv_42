146CgiProcess CgiHandler::startAsync(HttpRequest& request, int clientFd)
147{
148 int pipeIn[2];
149 int pipeOut[2];
150 if (pipe(pipeIn) < 0 || pipe(pipeOut) < 0)
151 throw std::runtime_error("CgiHandler: pipe() failed");
152
153 fcntl(pipeIn[1], F_SETFL, O_NONBLOCK);
154 fcntl(pipeOut[0], F_SETFL, O_NONBLOCK);
155
156 pid_t pid = fork();
157 if (pid < 0)
158 throw std::runtime_error("CgiHandler: fork() failed");
159
160 if (pid == 0)
161 {
162 dup2(pipeIn[0], STDIN_FILENO);
163 dup2(pipeOut[1], STDOUT_FILENO);
164 close(pipeIn[1]);
165 close(pipeOut[0]);
166
167 std::string resolvedPath = request.getResolvedPath();
168 std::string rootDir = resolvedPath.substr(0, resolvedPath.find_last_of('/'));
169 if (chdir(rootDir.c_str()) == -1)
170 _exit(EXIT_FAILURE);
171
172 char** envp = buildEnvp(request);
173 char* argv[] = { &resolvedPath[0], NULL };
174 execve(resolvedPath.c_str(), argv, envp);
175
176 freeEnvp(envp);
177 _exit(EXIT_FAILURE);
178 }
179
180 Signals::registerCgiProcess(pid);
181 close(pipeIn[0]);
182 close(pipeOut[1]);
183
184 const std::string& body = request.getBody();
185 if (!body.empty())
186 write(pipeIn[1], body.c_str(), body.size());
187 close(pipeIn[1]);
188
189 CgiProcess proc;
190 proc.pid = pid;
191 proc.in_fd = -1;
192 proc.out_fd = pipeOut[0];
193 proc.err_fd = -1;
194 proc.headersParsed = false;
195 proc.finished = false;
196 proc.chunked = false;
197 proc.startAt = time(NULL);
198 proc.deadline = proc.startAt + Signals::CGI_TIMEOUT_SEC;
199 proc.client_fd = clientFd;
200
201 fcntl(proc.out_fd, F_SETFL, O_NONBLOCK);
202
203 Logger::instance().log(DEBUG, "CGI: started async pid=" + toString(pid) +
204 " fd=" + toString(proc.out_fd));
205
206 return proc;
207}
