27void Router::resolve(HttpRequest& req, HttpResponse& res, const ServerConfig& config)
28{
29 //Handle parser-level errors before routing logic
30 if (req.getParseError() != ResponseStatus::OK)
31 {
32 Logger::instance().log(WARNING, "Router: Request parse error detected");
33 req.setRouteType(RouteType::Error);
34 res.setStatusCode(req.getParseError());
35 return ;
36 }
37
38 //Match the configuration block ("location") for the requested URI
39 const LocationConfig& loc = config.matchLocation(req.getUri());
40 const std::string index = loc.getHasIndexFiles() ? loc.getIndex() : "";
41 const std::string root = loc.getHasRoot() ? loc.getRoot() : config.getRoot();
42 const std::string cgiPath = loc.getCgiPath();
43
44 //Path traversal security check
45 if (hasParentTraversal(req.getUri()))
46 {
47 Logger::instance().log(WARNING, "Router: Path traversal attempt blocked: " + req.getUri());
48 req.setRouteType(RouteType::Error);
49 res.setStatusCode(ResponseStatus::Forbidden);
50 return ;
51 }
52
53 //Build filesystem path corresponding to request URI
54 computeResolvedPath(req, loc, config);
55
56 //Handle configured HTTP redirects
57 if (isRedirect(req, res, config))
58 {
59 Logger::instance().log(INFO, "Router: Route type = Redirect");
60 req.setRouteType(RouteType::Redirect);
61 return ;
62 }
63
64 //Handle CGI execution requests
65 if (isCgi(loc, req, res))
66 {
67 Logger::instance().log(INFO, "Router: Route type = CGI");
68 req.setRouteType(RouteType::CGI);
69 return ;
70 }
71
72 if (checkErrorStatus(req, res))
73 return ;
74
75 //Handle file uploads (POST/PUT)
76 if (isUpload(req, res, config))
77 {
78 Logger::instance().log(INFO, "Router: Route type = Upload");
79 req.setRouteType(RouteType::Upload);
80 return ;
81 }
82
83 if (checkErrorStatus(req, res))
84 return ;
85
86 //Handle AutoIndex directory listings
87 if (index.empty() && isAutoIndex(index, req, config))
88 {
89 Logger::instance().log(INFO, "Router: Route type = AutoIndex");
90 req.setRouteType(RouteType::AutoIndex);
91 return ;
92 }
93
94 if (checkErrorStatus(req, res))
95 return ;
96
97 //Handle DELETE requests explicitly
98 if (req.getMethod() == RequestMethod::DELETE)
99 {
100 req.setRouteType(RouteType::Delete);
101 return ;
102 }
103
104 // Handle static files (GET requests)
105 if (isStaticFile(index, req, res))
106 {
107 Logger::instance().log(INFO, "Router: Route type = StaticPage");
108
109 if (req.getMethod() != RequestMethod::GET)
110 {
111 Logger::instance().log(WARNING, "Router: Static file requested with invalid method");
112 req.setRouteType(RouteType::Error);
113 res.setStatusCode(ResponseStatus::MethodNotAllowed);
114 return ;
115 }
116
117 req.setRouteType(RouteType::StaticPage);
118 return ;
119 }
120
121 if (checkErrorStatus(req, res))
122 return ;
123
124 //Default case: route not found (404)
125 Logger::instance().log(WARNING, "Router: No matching route found (404)");
126 res.setStatusCode(ResponseStatus::NotFound);
127 req.setRouteType(RouteType::Error);
128}

146void Router::computeResolvedPath(HttpRequest& req,
147 const LocationConfig& location,
148 const ServerConfig& config)
149{
150 // Original request URI (e.g., "/uploads/file.txt")
151 const std::string uri = req.getUri();
152
153 // Determine which root will be used: CGI root, location root, or server root.
154 bool hasCgiRoot = false;
155 bool hasLocRoot = false;
156 std::string root;
157
158 if (!location.getCgiPath().empty())
159 {
160 hasCgiRoot = true;
161 root = location.getCgiPath();
162 }
163 else if (location.getHasRoot())
164 {
165 hasLocRoot = true;
166 root = location.getRoot();
167 }
168 else
169 {
170 root = config.getRoot();
171 }
172
173 // Normalize location path (remove trailing slash but keep "/" itself).
174 std::string locPath = location.getPath();
175 if (locPath.size() > 1 && locPath[locPath.size() - 1] == '/')
176 {
177 locPath.erase(locPath.size() - 1);
178 }
179
180 // Compute the "tail" portion to append to root.
181 // - If location has its own root/CGI root: strip the location prefix from the URI.
182 // - Otherwise: keep the full URI.
183 std::string tail;
184 if (hasCgiRoot || hasLocRoot)
185 {
186 if (startsWith(uri, locPath))
187 {
188 // Example: uri="/cgi-bin/hello.py", locPath="/cgi-bin" -> tail="/hello.py"
189 tail = uri.substr(locPath.size());
190 }
191 else
192 {
193 // Defensive fallback: config mismatch or a prior rewrite changed the URI.
194 tail = uri;
195 }
196 }
197 else
198 {
199 // Inherit server root: keep the entire URI (e.g., "/uploads/file.txt")
200 tail = uri;
201 }
202
203 // Remove leading slash from tail so joinPaths(root, tail) works consistently.
204 if (!tail.empty() && tail[0] == '/')
205 {
206 tail.erase(0, 1);
207 }
208
209 // Join root and tail. Expected behavior of joinPaths:
210 // - If 'tail' is empty, it should return 'root' unchanged.
211 // - It should avoid duplicating slashes.
212 std::string resolved = joinPaths(root, tail);
213
214 // If the resolved path is a directory and the location defines an index,
215 // append that index file.
216 struct stat st;
217 if (stat(resolved.c_str(), &st) == 0)
218 {
219 if (S_ISDIR(st.st_mode))
220 {
221 if (location.getHasIndexFiles())
222 {
223 resolved = joinPaths(resolved, location.getIndex());
224 }
225 }
226 }
227
228 // Store the final path in the request object.
229 req.setResolvedPath(resolved);
230
231 // Diagnostic log (useful to confirm mapping behavior).
232 Logger::instance().log(
233 DEBUG,
234 "Router::computeResolvedPath: uri=" + uri +
235 " locPath=" + locPath +
236 " root=" + root +
237 " -> resolved=" + resolved
238 );
239}

285bool Router::isUpload(HttpRequest& req, HttpResponse& res, const ServerConfig& config)
286{
287 const LocationConfig& location = config.matchLocation(req.getUri());
288 std::string uploadPath = location.getUploadPath();
289 const std::string& uri = req.getUri();
290
291 Logger::instance().log(DEBUG, "Router::isUpload comparing uri=" + uri + " uploadPath=" + uploadPath);
292
293 // Only POST or PUT methods are valid for upload
294 if (req.getMethod() != RequestMethod::POST && req.getMethod() != RequestMethod::PUT)
295 return (false);
296
297 // Uploads disabled at location level
298 if (!location.getUploadEnabled())
299 {
300 Logger::instance().log(WARNING, "Router::isUpload disabled for this location (403)");
301 return (false);
302 }
303
304 std::string basePath = uploadPath;
305 if (basePath.empty())
306 return (false);
307
308 // Handle relative upload paths
309 if (basePath[0] != '/')
310 basePath = config.getRoot() + "/" + basePath;
311
312 // Validate directory existence and write access
313 if (uri.find(location.getPath()) == 0)
314 {
315 struct stat s;
316 if (stat(basePath.c_str(), &s) != 0 || !S_ISDIR(s.st_mode))
317 {
318 Logger::instance().log(WARNING, "Router::isUpload directory missing: " + basePath);
319 res.setStatusCode(ResponseStatus::InternalServerError);
320 req.setRouteType(RouteType::Error);
321 return (false);
322 }
323 if (access(basePath.c_str(), W_OK) != 0)
324 {
325 Logger::instance().log(WARNING, "Router::isUpload path not writable: " + basePath);
326 res.setStatusCode(ResponseStatus::Forbidden);
327 req.setRouteType(RouteType::Error);
328 return (false);
329 }
330
331 // Directory OK
332 req.setRouteType(RouteType::Upload);
333 return (true);
334 }
335
336 return (false);
337}

342bool Router::isAutoIndex(const std::string& index, HttpRequest& req, const ServerConfig& config)
343{
344 const LocationConfig& location = config.matchLocation(req.getUri());
345
346 bool autoIndexEnabled = config.getAutoindex();
347
348 // Location-specific directive overrides global setting
349 if (location.getHasAutoIndex())
350 autoIndexEnabled = location.getAutoindex();
351
352 if (!autoIndexEnabled)
353 return (false);
354
355 std::string path = req.getResolvedPath();
356 struct stat s;
357
358 // AutoIndex only applies to directories without index files
359 if (stat(path.c_str(), &s) == 0 && S_ISDIR(s.st_mode))
360 {
361 if (path[path.length() - 1] != '/')
362 path += '/';
363
364 path += index;
365
366 struct stat sIndex;
367 if (stat(path.c_str(), &sIndex) != 0 || !S_ISREG(sIndex.st_mode))
368 {
369 Logger::instance().log(DEBUG, "Router::isAutoIndex enabled for directory: " + req.getResolvedPath());
370 return (true);
371 }
372 }
